pols_month_df = 
    read_csv("fivethirtyeight_datasets/pols-month.csv") |> 
    janitor::clean_names() |> 
    separate(mon, into = c("year", "month", "day"), convert = TRUE) |> 
    mutate(month = month.name[month],
      president = prez_gop - prez_dem,
      president = case_when(
          president > 0 ~ "gop",
          president < 0 ~ "dem"),
      year = as.numeric(year)) |> 
    select(-prez_gop, -prez_dem, -day)
